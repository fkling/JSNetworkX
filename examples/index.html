Undirected weighted graph
                
var G1 = jsnx.Graph();

G1.add_weighted_edges_from([[2,3,10]]);

G1.add_star([3,4,5,6], {weight: 5}); 

G1.add_star([2,1,0,-1], {weight: 3});

jsnx.draw(G1, {
    element: '#chart1',  
    weighted: true, 
    edge_style: {
        'stroke-width': 10
    }
}, true);
Directed graph with labels
                
var G2 = jsnx.DiGraph();

G2.add_nodes_from([1,2,3,4,5,[9,{color: '#008A00'}]], {color: '#0064C7'});

G2.add_cycle([1,2,3,4,5]);

G2.add_edges_from([[1,9], [9,1]]);

jsnx.draw(G2, {
    element: '#chart2', 
    with_labels: true, 
    node_style: {
        fill: function(d) { 
            return d.data.color; 
        }
    }, 
    label_style: {fill: 'white' }
}, true);
NetworkX example graph
                
var G3 = jsnx.Graph();

G3.add_nodes_from([1,2,3,4], {group:0});
G3.add_nodes_from([5,6,7], {group:1});
G3.add_nodes_from([8,9,10,11], {group:2});

G3.add_path([1,2,5,6,7,8,11]);
G3.add_edges_from([[1,3],[1,4],[3,4],[2,3],[2,4],[8,9],[8,10],[9,10],[11,10],[11,9]]);

var color = d3.scale.category20();
jsnx.draw(G3, {
    element: '#chart3',
    layout_attr: {
        charge: -120,
        linkDistance: 20
    },
    node_attr: {
        r: 5,
        title: function(d) { return d.label;}
    },
    node_style: {
        fill: function(d) { 
            return color(d.data.group); 
        },
        stroke: 'none'
    },
    edge_style: {
        fill: '#999'
    }
}, true);
